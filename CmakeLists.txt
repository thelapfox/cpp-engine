cmake_minimum_required(VERSION 3.10)

project(MyGameEngine)

# Set the output directory for the binary files
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

# Add subdirectories for external libraries
add_subdirectory(external/lib1)
add_subdirectory(external/lib2)
add_subdirectory(external/lib3)

# Add the include directories for the external libraries
include_directories(
    ${PROJECT_SOURCE_DIR}/external/lib1/include
    ${PROJECT_SOURCE_DIR}/external/lib2/include
    ${PROJECT_SOURCE_DIR}/external/lib3/include
)

# Add the source files for your game engine
add_executable(engine 
    src/main.cpp 
    src/engine.cpp
)

# Link against the external libraries
target_link_libraries(engine 
    lib1
    lib2
    lib3
)
